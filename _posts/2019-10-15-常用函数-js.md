### 获取query值得方法

```javascript
  /*获取query值得方法*/
    function getQuery(queryName) {
        let url = window.location.search.substr(1)
        let query = ""
        url.split('&').forEach(item => {
            if (item.split('=')[0] == queryName) {
                query = item.split('=')[1]
            }
        })
        return query
    }

```
